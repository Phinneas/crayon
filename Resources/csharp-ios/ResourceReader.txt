using System;

namespace %%%PROJECT_ID%%%
{
	public static class ResourceReader
	{
		public static GameViewController GVC { get; set; }

		public static string ReadByteCodeFile()
		{
			return GVC.LoadResource("ByteCode", "txt");
		}

		public static object ReadImageFile(string path, bool isGeneratedImage)
		{
			int dot = path.LastIndexOf('.');
			if (dot == -1) return null;
			string type = path.Substring(dot + 1);
			path = path.Substring(0, dot);
			return GVC.LoadResource(path, type);
		}
	}
}
