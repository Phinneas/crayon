package org.crayonlang.interpreter;

import java.util.ArrayList;
import java.util.Collection;
import java.util.HashMap;
import java.util.List;

public final class LibraryLoader {
  public static Class<?> getLibClass(String name, String version) {
    try {
      return Class.forName("org.crayonlang.libraries." + name.toLowerCase() + ".LibraryWrapper");
    } catch (Exception e) {
      throw new RuntimeException("Library could not be found.", e);
	}
  }
}
